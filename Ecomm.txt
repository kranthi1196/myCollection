                 - setSubject(username)
                 - setIssuedAt(new Date())
                 - setExpiration(new Date(System - currentTimeMillis() + 1000 * 60 * 60 * 10)) // 10 hours
                 - signWith(SignatureAlgorithm - HS256, SECRET_KEY)
                 - compact();
    }

    public Boolean validateToken(String token) {

            return true;
        } catch (Exception e) {
            return false;
        }
    }

        return claims getSubject();
    }
